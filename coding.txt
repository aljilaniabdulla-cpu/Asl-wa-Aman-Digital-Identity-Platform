// Button Component
const Button = ({ children, variant = 'primary', className = '', onClick, ...props }) => {
  const baseStyle = "px-6 py-2 rounded-lg font-medium transition-all duration-300 flex items-center justify-center gap-2";
  const variants = {
    primary: "bg-emerald-800 text-white hover:bg-emerald-900 shadow-lg hover:shadow-xl",
    secondary: "bg-amber-600 text-white hover:bg-amber-700 shadow-md",
    outline: "border-2 border-emerald-800 text-emerald-800 hover:bg-emerald-50",
    ghost: "text-gray-600 hover:bg-gray-100"
  };

  return (
    <button 
      className={`${baseStyle} ${variants[variant]} ${className}`} 
      onClick={onClick}
      {...props}
    >
      {children}
    </button>
  );
};

             <div className="space-y-6">
               {requests.map((req) => (
                 <div key={req.id} className="bg-white rounded-xl shadow-sm p-6 border border-gray-100">
                    <div className="flex flex-col md:flex-row justify-between items-start md:items-center mb-6 gap-4">
                       <div>
                         <h3 className="font-bold text-lg text-gray-800">{req.type}</h3>
                         <p className="text-sm text-gray-500">رقم الطلب: {req.id}</p>
                       </div>
                       {getStatusBadge(req.status)}
                    </div>
                    
                    {/* Timeline */}
                    <div className="relative pt-4">
                      <div className="absolute top-1/2 left-0 w-full h-1 bg-gray-100 -z-10 transform -translate-y-1/2 rounded"></div>
                      <div className="flex justify-between text-sm">
                        <div className="flex flex-col items-center gap-2">
                          <div className="w-8 h-8 rounded-full bg-emerald-600 text-white flex items-center justify-center"><CheckCircle size={16} /></div>
                          <span className="text-emerald-800 font-medium">تم الإرسال</span>
                        </div>
                        <div className="flex flex-col items-center gap-2">
                          <div className={`w-8 h-8 rounded-full flex items-center justify-center ${req.status !== 'pending' ? 'bg-emerald-600 text-white' : 'bg-gray-200'}`}>
                            {req.status === 'processing' && <Activity size={16} className="animate-pulse" />} 
                            {req.status === 'completed' && <CheckCircle size={16} />}
                          </div>
                          <span className={`${req.status === 'processing' ? 'text-blue-600 font-bold' : 'text-gray-500'}`}>تحليل AI</span>
                        </div>
                        <div className="flex flex-col items-center gap-2">
                          <div className={`w-8 h-8 rounded-full flex items-center justify-center ${req.status === 'completed' ? 'bg-emerald-600 text-white' : 'bg-gray-200'}`}>
                             {req.status === 'completed' && <CheckCircle size={16} />}
                          </div>
                          <span className="text-gray-500">الموافقة</span>
                        </div>
                      </div>
                    </div>

                    {req.status === 'processing' && (
                      <div className="mt-6 p-4 bg-blue-50 text-blue-800 rounded-lg text-sm flex items-center gap-3">
                        <Activity size={18} />
                         جاري التحقق من صحة الوثيقة ومطابقتها مع السجلات التاريخية باستخدام الذكاء الاصطناعي...
                      </div>
                    )}
                 </div>
               ))}
             </div>
          )}

        </div>
      </main>
    </div>
  );
};
